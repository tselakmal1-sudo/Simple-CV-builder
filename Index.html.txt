            init() {
                this.loadState();
                this.setupEventListeners();
                this.updateAuthUI();
                this.updateLanguage();
                this.navigate('home');
                this.preloadDemoData();
            },

            loadState() {
                const saved = localStorage.getItem('appState');
                if (saved) {
                    const parsed = JSON.parse(saved);
                    this.state.cvs = parsed.cvs || [];
                    this.state.coverLetters = parsed.coverLetters || [];
                }
            },

            saveState() {
                localStorage.setItem('appState', JSON.stringify({
                    cvs: this.state.cvs,
                    coverLetters: this.state.coverLetters
                }));
            },

            setupEventListeners() {
                window.addEventListener('popstate', () => {
                    const page = location.hash.slice(1) || 'home';
                    this.navigate(page, false);
                });
            },

            navigate(page, pushState = true) {
                if (pushState) history.pushState(null, '', `#${page}`);
                document.querySelectorAll('.page').forEach(p => p.classList.add('hidden'));
                const target = document.getElementById(`page-${page}`);
                if (target) {
                    target.classList.remove('hidden');
                    window.scrollTo(0, 0);
                }
                
                if (page === 'dashboard') this.renderDashboard();
                if (page === 'builder') this.renderBuilder();
                if (page === 'jobmatcher') this.initJobMatcher();
                if (page === 'coverletter') this.initCoverLetter();
                
                this.updateLanguage();
            },

            toggleLanguage() {
                this.state.lang = this.state.lang === 'en' ? 'si' : 'en';
                localStorage.setItem('lang', this.state.lang);
                document.getElementById('lang-label').textContent = this.state.lang.toUpperCase();
                this.updateLanguage();
            },

            updateLanguage() {
                const t = this.translations[this.state.lang];
                document.querySelectorAll('[data-i18n]').forEach(el => {
                    const key = el.getAttribute('data-i18n');
                    if (t[key]) el.textContent = t[key];
                });
            },

            updateAuthUI() {
                const authButtons = document.getElementById('auth-buttons');
                const userMenu = document.getElementById('user-menu');
                if (this.state.user) {
                    authButtons.classList.add('hidden');
                    userMenu.classList.remove('hidden');
                    document.getElementById('user-name').textContent = this.state.user.name;
                    document.getElementById('dash-username').textContent = this.state.user.name;
                } else {
                    authButtons.classList.remove('hidden');
                    userMenu.classList.add('hidden');
                }
            },

            switchAuthTab(tab) {
                document.getElementById('tab-login').className = tab === 'login' ? 'flex-1 py-2 rounded-md text-sm font-medium transition bg-white shadow-sm text-gray-900' : 'flex-1 py-2 rounded-md text-sm font-medium transition text-gray-500';
                document.getElementById('tab-register').className = tab === 'register' ? 'flex-1 py-2 rounded-md text-sm font-medium transition bg-white shadow-sm text-gray-900' : 'flex-1 py-2 rounded-md text-sm font-medium transition text-gray-500';
                document.getElementById('login-form').classList.toggle('hidden', tab !== 'login');
                document.getElementById('register-form').classList.toggle('hidden', tab !== 'register');
            },

            handleLogin(e) {
                e.preventDefault();
                this.state.user = { name: 'Demo User', email: 'demo@example.com' };
                localStorage.setItem('user', JSON.stringify(this.state.user));
                this.updateAuthUI();
                this.navigate('dashboard');
            },

            handleRegister(e) {
                e.preventDefault();
                const name = document.getElementById('reg-name').value;
                const email = document.getElementById('reg-email').value;
                this.state.user = { name, email };
                localStorage.setItem('user', JSON.stringify(this.state.user));
                this.updateAuthUI();
                this.navigate('dashboard');
            },

            logout() {
                this.state.user = null;
                localStorage.removeItem('user');
                this.updateAuthUI();
                this.navigate('home');
            },

                    this.updateAuthUI();
                    
                    // Pre-fill some data
                    document.getElementById('field-name').value = 'Alex Fernando';
                    document.getElementById('field-title').value = 'Senior Software Engineer';
                    document.getElementById('field-email').value = 'alex@example.com';
                    document.getElementById('field-linkedin').value = 'linkedin.com/in/alexfernando';

                    btn.innerHTML = '<i class="fab fa-linkedin"></i> Import from LinkedIn';
                }, 1500);
            },

            startBuilding() {
                this.state.currentCV = {
                    personal: {}, education: [], experience: [], skills: [], languages: [],
                    template: 'modern', color: '#4F46E5',
                    settings: { atsFriendly: false, includeQR: false, includePhoto: true },
                    photoData: null
                };
                this.state.currentStep = 1;
                this.navigate('builder');
            },

            selectTemplateAndBuild(template) {
                this.startBuilding();
                this.state.currentCV.template = template;
                setTimeout(() => this.selectTemplate(template), 100);
            },

            renderBuilder() {
                this.updateStepIndicators();
                document.querySelectorAll('.form-step').forEach((el, i) => {
                    el.classList.toggle('hidden', i + 1 !== this.state.currentStep);
                });
                
                document.getElementById('btn-prev').classList.toggle('hidden', this.state.currentStep === 1);
                document.getElementById('btn-next').classList.toggle('hidden', this.state.currentStep === 5);
                document.getElementById('btn-generate').classList.toggle('hidden', this.state.currentStep !== 5);
                
                // Update button text
                const nextBtn = document.getElementById('btn-next');
                if (this.state.currentStep === 4) {
                    nextBtn.querySelector('span').textContent = this.state.lang === 'si' ? 'ටෙම්ප්ලේට් වෙත' : 'To Template';
                } else {
                    nextBtn.querySelector('span').textContent = this.translations[this.state.lang].btn_next;
                }
                
                this.renderPreview();
                
                // Initialize dynamic lists
                if (this.state.currentStep === 2 && this.state.currentCV.education.length === 0) this.addEducation();
                if (this.state.currentStep === 4 && this.state.currentCV.experience.length === 0) this.addExperience();
            },

            updateStepIndicators() {
                for (let i = 1; i <= 5; i++) {
                    const indicator = document.getElementById(`step-${i}-indicator`);
                    const line = document.getElementById(`progress-line-${i}`);
                    
                    if (i < this.state.currentStep) {
                        indicator.className = 'w-10 h-10 rounded-full border-2 border-green-500 bg-green-500 text-white flex items-center justify-center font-bold step-indicator step-completed';
                        indicator.innerHTML = '<i class="fas fa-check"></i>';
                        if (line) line.style.width = '100%';
                    } else if (i === this.state.currentStep) {
                        indicator.className = 'w-10 h-10 rounded-full border-2 border-primary bg-primary text-white flex items-center justify-center font-bold step-indicator step-active';
                        indicator.textContent = i;
                        if (line) line.style.width = '0%';
                    } else {
                        indicator.className = 'w-10 h-10 rounded-full border-2 border-gray-300 text-gray-400 flex items-center justify-center font-bold step-indicator';
                        indicator.textContent = i;
                        if (line) line.style.width = '0%';
                    }
                }
            },

            nextStep() {
                if (this.state.currentStep < 5) {
                    this.saveCurrentStepData();
                    this.state.currentStep++;
                    this.renderBuilder();
                }
            },

            prevStep() {
                if (this.state.currentStep > 1) {
                    this.saveCurrentStepData();
                    this.state.currentStep--;
                    this.renderBuilder();
                }
            },

            saveCurrentStepData() {
                if (this.state.currentStep === 1) {
                    this.state.currentCV.personal = {
                        name: document.getElementById('field-name').value,
                        title: document.getElementById('field-title').value,
                        email: document.getElementById('field-email').value,
                        phone: document.getElementById('field-phone').value,
                        linkedin: document.getElementById('field-linkedin').value,
                        portfolio: document.getElementById('field-portfolio').value,
                        address: document.getElementById('field-address').value,
                        summary: document.getElementById('field-summary').value
                    };
                }
            },

            handlePhotoUpload(e) {

            darkenColor(hex, percent) {
                const num = parseInt(hex.replace('#', ''), 16);
                const amt = Math.round(2.55 * percent);
                const R = (num >> 16) - amt;
                const G = (num >> 8 & 0x00FF) - amt;
                const B = (num & 0x0000FF) - amt;
                return '#' + (0x1000000 + (R < 255 ? R < 1 ? 0 : R : 255) * 0x10000 + (G < 255 ? G < 1 ? 0 : G : 255) * 0x100 + (B < 255 ? B < 1 ? 0 : B : 255)).toString(16).slice(1);
            },

            calculateATSScore() {
                let score = 60;
                const { personal, education, experience, skills } = this.state.currentCV;
                
                if (personal.name && personal.email && personal.title) score += 10;
                if (personal.summary && personal.summary.length > 50) score += 5;
                if (education.length > 0) score += 5;
                if (experience.length > 0) score += 10;
                if (skills.length >= 5) score += 10;
                if (this.state.currentCV.settings.atsFriendly) score += 10;
                if (personal.linkedin || personal.portfolio) score += 5;
                
                return Math.min(98, score);
            },

            generateCV() {
                this.collectFormData();
                
                // Show AI modal
                const modal = document.getElementById('ai-modal');
                modal.classList.remove('hidden');
                
                const steps = ['ai-step-1', 'ai-step-2', 'ai-step-3', 'ai-step-4'];
                const progress = document.getElementById('ai-progress');
                
                let currentStep = 0;
                const interval = setInterval(() => {
                    if (currentStep > 0) {
                        const prev = document.getElementById(steps[currentStep - 1]);
                        prev.querySelector('i').className = 'fas fa-check text-green-500';
                        prev.querySelector('i').classList.remove('fa-spin');
                    }
                    
                    if (currentStep < steps.length) {
                        const curr = document.getElementById(steps[currentStep]);
                        curr.querySelector('i').className = 'fas fa-circle-notch fa-spin text-primary';
                        curr.classList.remove('opacity-40');
                        progress.style.width = ((currentStep + 1) / steps.length * 100) + '%';
                    }
                    
                    currentStep++;
                    
                    if (currentStep > steps.length) {
                        clearInterval(interval);
                        setTimeout(() => {
                            modal.classList.add('hidden');
                            this.saveCV();
                            this.navigate('preview');
                        }, 500);
                    }
                }, 800);
            },

            async downloadPDF() {
                const element = document.getElementById('full-cv-preview').firstElementChild || document.getElementById('cv-preview-container').firstElementChild;
                if (!element) return;
                
                try {
                    const canvas = await html2canvas(element, {
                        scale: 2,
                        useCORS: true,
                        allowTaint: true,
                        backgroundColor: '#ffffff'
                    });
                    
                    const imgData = canvas.toDataURL('image/png');
                    const pdf = new jspdf.jsPDF('p', 'mm', 'a4');
                    const pdfWidth = pdf.internal.pageSize.getWidth();
                    const pdfHeight = (canvas.height * pdfWidth) / canvas.width;
                    
                    pdf.addImage(imgData, 'PNG', 0, 0, pdfWidth, pdfHeight);
                    pdf.save(`${this.state.currentCV.personal.name || 'CV'}_CV.pdf`);
                } catch (err) {
                    console.error('PDF generation failed:', err);
                    alert('PDF generation failed. Please try again.');
                }
            },

                let userSkills = manualSkills;
                if (cvIndex !== '') {
                    const cv = this.state.cvs[cvIndex];
                    userSkills = [...cv.skills.map(s => s.toLowerCase()), ...manualSkills];
                }
                
                if (userSkills.length === 0) {
                    alert(this.state.lang === 'si' ? 'කරුණාකර කුසලතා ඇතුළත් කරන්න' : 'Please enter some skills');
                    return;
                }
                
                const results = this.jobsDB.map(job => {
                    const jobSkills = job.skills.map(s => s.toLowerCase());
                    const matched = jobSkills.filter(s => userSkills.some(us => us.includes(s) || s.includes(us)));
                    const matchPercent = Math.round((matched.length / jobSkills.length) * 100);
                    return { ...job, matchPercent, matchedSkills: matched };
                }).sort((a, b) => b.matchPercent - a.matchPercent);
                
                const container = document.getElementById('job-results');
                container.innerHTML = results.map(job => `
                    <div class="bg-white rounded-xl shadow-sm border border-gray-100 p-6 hover:shadow-md transition">
                        <div class="flex items-start justify-between mb-4">
                            <div>
                                <h3 class="font-bold text-lg text-gray-900">${job.title}</h3>
                                <p class="text-gray-600">${job.company} • ${job.location}</p>
                            </div>
                            <div class="text-right">
                                <div class="text-2xl font-bold ${job.matchPercent >= 70 ? 'text-green-600' : job.matchPercent >= 40 ? 'text-yellow-600' : 'text-gray-500'}">${job.matchPercent}%</div>
                                <div class="text-xs text-gray-500">Match</div>
                            </div>
                        </div>
                        
                        <div class="flex flex-wrap gap-2 mb-4">
                            ${job.skills.map(s => `
                                <span class="px-2 py-1 rounded-md text-xs font-medium ${job.matchedSkills.includes(s.toLowerCase()) ? 'bg-green-100 text-green-700' : 'bg-gray-100 text-gray-600'}">
                                    ${s} ${job.matchedSkills.includes(s.toLowerCase()) ? '<i class="fas fa-check ml-1"></i>' : ''}
                                </span>
                            `).join('')}

                                Apply Now
                            </button>
                        </div>

                `).join('');
            },
